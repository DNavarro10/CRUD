CREATE TABLE `estudiantes`.`alumnos` ( `cedula` INT(10) NOT NULL AUTO_INCREMENT , `nombre` VARCHAR(40) NOT NULL , `primer_apellido` VARCHAR(20) NOT NULL , `segun_apellido` VARCHAR(20) NOT NULL , `edad` TINYINT NOT NULL , `telefono` INT(10) NOT NULL , `direccion` VARCHAR(200) NOT NULL , PRIMARY KEY (`cedula`)) ENGINE = InnoDB;
